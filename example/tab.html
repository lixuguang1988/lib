.jtab-nav, .jtab-panels, jtab-more[可有]三个类名是必须的，操作的是他们的第一级子节点
.jtab-nav里面的超链接不是必须的
<div class="example">
	<ul class="jtab-nav">
		<li class="current"><a href="http://www.baidu.com">Baidu</a></li>
		<li><a href="http://www.alibaba.com">Alibaba</a></li>
		<li><a href="http://www.qq.com">Tencent</a></li>
	</ul>
	<div class="jtab-more">
		<a href="http://www.baidu.com">more>></a>
		<a class="hide" href="http://www.alibaba.com">more>></a>
		<a class="hide" href="http://www.qq.com">more>></a>
	</div>
	<div class="jtab-panels">
		<div class="jtab-panel">上海市密云路幼儿园管理员组1</div>
		<div class="jtab-panel hide">上海市密云路幼儿园管理员组2</div>
		<div class="jtab-panel hide">上海市密云路幼儿园管理员组2</div>
	</div>
</div>
<script>
       $('.example').jtab({
			trigger : "mouseover", //触发事件[mouseover, click]
			currentClass : "current", //tab-nav上的当前类
			preventDefault : true  //阻止默认事件 [true, false]
      	 });
</script>
